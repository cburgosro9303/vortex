5/// Response struct compatible with Spring Cloud Config Server v2.
6///
7/// This serves as a Data Transfer Object (DTO) to ensure the JSON output
8/// matches exactly what Spring Boot clients expect.
9#[derive(Debug, Clone, Serialize, Deserialize)]
10#[serde(rename_all = "camelCase")]
11pub struct SpringConfigResponse {
12    pub name: String,
13    pub profiles: Vec<String>,
14
15    #[serde(skip_serializing_if = "Option::is_none")]
16    pub label: Option<String>,
17
18    #[serde(skip_serializing_if = "Option::is_none")]
19    pub version: Option<String>,
20
21    #[serde(skip_serializing_if = "Option::is_none")]
22    pub state: Option<String>,
23
24    #[serde(rename = "propertySources")]
25    pub property_sources: Vec<SpringPropertySource>,
26}

28/// Property source in Spring format (flattened keys).
29#[derive(Debug, Clone, Serialize, Deserialize)]
30pub struct SpringPropertySource {
31    pub name: String,
32    pub source: IndexMap<String, ConfigValue>,
33}

35impl SpringConfigResponse {
36    /// Creates a new builder/response.

38        Self {
39            name: name.into(),
40            profiles,
41            label: None,
42            version: None,
43            state: None,
44            property_sources: Vec::new(),

48    pub fn with_label(mut self, label: impl Into<String>) -> Self {
49        self.label = Some(label.into());
50        self
51    }
52
53    pub fn with_version(mut self, version: impl Into<String>) -> Self {
54        self.version = Some(version.into());
55        self
56    }
57
58    pub fn add_source(mut self, source: SpringPropertySource) -> Self {

61    }

77/// Flattens a hierarchical ConfigMap into a flat map with dot-notation keys.
78///
79/// Example:
80/// {"server": {"port": 80}} -> {"server.port": 80}
81pub fn flatten_config_map(config: &ConfigMap) -> IndexMap<String, ConfigValue> {
82    let mut flat_map = IndexMap::new();
83    for (key, value) in config.as_inner() {
84        flatten_value(key, value, &mut flat_map);
85    }
86    flat_map
87}
88
89fn flatten_value(prefix: &str, value: &ConfigValue, target: &mut IndexMap<String, ConfigValue>) {
90    match value {
91        ConfigValue::Object(map) => {
92            for (curr_key, curr_val) in map {
93                let new_key = format!("{}.{}", prefix, curr_key);
94                flatten_value(&new_key, curr_val, target);
95            }
96        },
97        // For Spring compatibility, arrays are often treated as values or indexed keys.
98        // Here we treat array as a value (leaf) as per our planning decision.
99        _ => {
100            target.insert(prefix.to_string(), value.clone());
101        },
102    }
103}

108    use crate::config::ConfigMap;

110    #[test]
111    fn test_flattening_logic() {
112        let json = r#"{
113            "server": {
114                "port": 8080,
115                "ssl": {
116                    "enabled": true
117                }
118            },
119            "app": "test"
120        }"#;
121        let config = ConfigMap::from_json(json).unwrap();
122        let flat = flatten_config_map(&config);
123
124        assert_eq!(flat.get("server.port").unwrap().as_i64(), Some(8080));
125        assert_eq!(
126            flat.get("server.ssl.enabled").unwrap().as_bool(),
127            Some(true)
128        );
129        assert_eq!(flat.get("app").unwrap().as_str(), Some("test"));
130
131        // Ensure intermediate keys shouldn't exist as separate entries
132        assert!(flat.get("server").is_none());
133        assert!(flat.get("server.ssl").is_none());
134    }

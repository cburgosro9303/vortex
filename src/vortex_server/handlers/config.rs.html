23/// Handler for GET /{app}/{profile} with state.
24#[instrument(skip_all, fields(app = %path.app, profile = %path.profile))]
25pub async fn get_config(
26    State(state): State<AppState>,
27    Path(path): Path<AppProfilePath>,
28    Query(_query): Query<ConfigQuery>,
29    AcceptFormat(format): AcceptFormat,
30) -> Result<Response, AppError> {
31    path.validate().map_err(AppError::BadRequest)?;
32
33    let profiles = path.profiles();
34
35    tracing::info!("Fetching config for {}/{:?}", path.app, profiles);
36
37    // Use default label for this endpoint
38    let label = state.config_source().default_label().to_string();
39
40    // Get configuration (with cache if enabled)
41    let response = match state.cache() {
42        Some(cache) => {
43            // Create cache key
44            let cache_key = CacheKey::new(&path.app, profiles.join(","), &label);
45
46            // Try to get from cache or fetch from backend
47            cache
48                .get_or_insert_with(cache_key, || {
49                    let config_source = state.config_source();
50                    let app = path.app.clone();
51                    let profiles = profiles.clone();
52                    async move { fetch_config(config_source, &app, profiles, &label).await }
53                })
54                .await
55                .map_err(|e: CacheError| AppError::Internal(e.to_string()))?
56        },
57        None => {
58            // No cache, fetch directly
59            let response = fetch_config(state.config_source(), &path.app, profiles, &label).await?;
60            Arc::new(response)
61        },
62    };
63
64    to_format(response.as_ref(), format).map_err(|e| AppError::Internal(format!("{:?}", e)))
65}

67/// Handler for GET /{app}/{profile}/{label} with state.
68#[instrument(skip_all, fields(
69    app = %path.app,
70    profile = %path.profile,
71    label = %path.label
72))]
73pub async fn get_config_with_label(
74    State(state): State<AppState>,
75    Path(path): Path<AppProfileLabelPath>,
76    Query(query): Query<ConfigQuery>,
77    AcceptFormat(format): AcceptFormat,
78) -> Result<Response, AppError> {
79    path.validate().map_err(AppError::BadRequest)?;
80
81    let profiles = path.profiles();
82    let label = path.sanitized_label();
83
84    tracing::info!(
85        use_default_label = query.use_default_label,
86        "Fetching config with label"
87    );
88
89    // Validate dangerous characters in label
90    validate_label(&label)?;
91
92    // Get configuration (with cache if enabled)
93    let response =
94        match state.cache() {
95            Some(cache) => {
96                // Create cache key
97                let cache_key = CacheKey::new(&path.app, profiles.join(","), &label);
98
99                // Try to get from cache or fetch from backend
100                match cache
101                    .get_or_insert_with(cache_key.clone(), || {
102                        let config_source = state.config_source();
103                        let app = path.app.clone();
104                        let profiles = profiles.clone();
105                        let label = label.clone();
106                        async move { fetch_config(config_source, &app, profiles, &label).await }
107                    })
108                    .await
109                {
110                    Ok(response) => response,
111                    Err(_) if query.use_default_label => {
112                        // Fallback to default label
113                        let default_label = state.config_source().default_label().to_string();
114                        tracing::info!(
115                            original_label = %label,
116                            default_label = %default_label,
117                            "Label not found, falling back to default"
118                        );
119
120                        let fallback_key =
121                            CacheKey::new(&path.app, profiles.join(","), &default_label);
122                        cache
123                            .get_or_insert_with(fallback_key, || {
124                                let config_source = state.config_source();
125                                let app = path.app.clone();
126                                let profiles = profiles.clone();
127                                async move {
128                                    fetch_config(config_source, &app, profiles, &default_label)
129                                        .await
130                                }
131                            })
132                            .await
133                            .map_err(|e: CacheError| AppError::Internal(e.to_string()))?
134                    },
135                    Err(e) => return Err(AppError::Internal(e.to_string())),
136                }
137            },
138            None => {
139                // No cache, fetch directly with fallback logic
140                let response =
141                    match fetch_config(state.config_source(), &path.app, profiles.clone(), &label)
142                        .await
143                    {
144                        Ok(response) => response,
145                        Err(_) if query.use_default_label => {
146                            let default_label = state.config_source().default_label();
147                            tracing::info!(
148                                original_label = %label,
149                                default_label = %default_label,
150                                "Label not found, falling back to default"
151                            );
152                            fetch_config(state.config_source(), &path.app, profiles, default_label)
153                                .await
154                                .map_err(|e| AppError::Internal(e.to_string()))?
155                        },
156                        Err(e) => return Err(AppError::Internal(e.to_string())),
157                    };
158                Arc::new(response)
159            },
160        };
161
162    to_format(response.as_ref(), format).map_err(|e| AppError::Internal(format!("{:?}", e)))
163}

165/// Converts a ConfigValue to serde_json::Value.
166fn config_value_to_json(value: &vortex_git::vortex_core::ConfigValue) -> serde_json::Value {
167    use vortex_git::vortex_core::ConfigValue;
168
169    match value {
170        ConfigValue::Null => serde_json::Value::Null,
171        ConfigValue::Bool(b) => serde_json::Value::Bool(*b),
172        ConfigValue::Integer(i) => serde_json::Value::Number((*i).into()),
173        ConfigValue::Float(f) => serde_json::Number::from_f64(f.into_inner())
174            .map(serde_json::Value::Number)
175            .unwrap_or(serde_json::Value::Null),
176        ConfigValue::String(s) => serde_json::Value::String(s.clone()),
177        ConfigValue::Array(arr) => {
178            serde_json::Value::Array(arr.iter().map(config_value_to_json).collect())
179        },
180        ConfigValue::Object(obj) => serde_json::Value::Object(
181            obj.iter()
182                .map(|(k, v)| (k.clone(), config_value_to_json(v)))
183                .collect(),
184        ),
185    }
186}

188/// Validates that the label does not contain dangerous characters.
189fn validate_label(label: &str) -> Result<(), AppError> {
190    // Prevent path traversal
191    if label.contains("..") {
192        return Err(AppError::BadRequest(
193            "Label cannot contain '..'".to_string(),
194        ));
195    }
196
197    // Prevent control characters
198    if label.chars().any(|c| c.is_control()) {
199        return Err(AppError::BadRequest(
200            "Label cannot contain control characters".to_string(),
201        ));
202    }
203
204    Ok(())
205}

207/// Fetches configuration from the backend and converts it to ConfigResponse.
208async fn fetch_config(
209    config_source: &dyn vortex_git::ConfigSource,
210    app: &str,
211    profiles: Vec<String>,
212    label: &str,
213) -> Result<ConfigResponse, CacheError> {
214    // Create query for the config source
215    let git_query = GitConfigQuery::new(app, profiles.clone()).with_label_set(label);
216
217    // Fetch from the config source
218    let result = config_source
219        .fetch(&git_query)
220        .await
221        .map_err(|e| CacheError::FetchError(e.to_string()))?;
222
223    // Convert to response format
224    Ok(ConfigResponse {
225        name: result.name().to_string(),
226        profiles: result.profiles().to_vec(),
227        label: Some(result.label().to_string()),
228        version: result.version().map(String::from),
229        state: result.state().map(String::from),
230        property_sources: result
231            .property_sources()
232            .iter()
233            .map(|ps| PropertySourceResponse {
234                name: ps.name.clone(),
235                source: ps
236                    .config
237                    .as_inner()
238                    .iter()
239                    .map(|(k, v)| (k.clone(), config_value_to_json(v)))
240                    .collect(),
241            })
242            .collect(),
243    })
244}
